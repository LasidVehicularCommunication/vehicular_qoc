56 ICRMessage * MessageManager::sendIcarMessageService(int64_t sourceId, int64_t nextId, int64_t destinationId, double timetoLiveApp, unsigned int pMsgType)
57 {
58 
59  ICRMessage * wsm = NULL;
60  wsm = new ICRMessage();
61 
62  //prepareing icmessage
63  wsm->setMsgType(pMsgType);
64  wsm->setNumMsg(++this->oIcarQoc->seqMsg);
65  wsm->setName("data");
66 
67  //User priority with which this packet was sent (note the AC mapping rules in Mac1609_4::mapUserPriority)
68  wsm->setUserPriority(7);
69 
70  //Unique number to identify the service
71  wsm->setPsid(-1);
72 
73  //populateWSM(wsm);
74  wsm->setKind(pMsgType);
75 
76  //Channel Number on which this packet was sent
77  wsm->setChannelNumber(static_cast<int>(Channel::cch));
78 
79  wsm->setMsgLifeTime(simTime() + timetoLiveApp);
80  wsm->setTimestamp(simTime());
81  wsm->setHopNumber(1);
82 
83  wsm->setTransmissorNode(prepareICRNode(oIcarQoc->oIcarContext->myData));
84 
85  if (oIcarQoc->oIcarContext->myData->getId()==sourceId || sourceId==-1){
86  wsm->setSourceNode(prepareICRNode(oIcarQoc->oIcarContext->myData));
87  }else {
88  wsm->setSourceNode(prepareICRNode(oIcarQoc->oIcarContext->agentGroup->getRemoteAgent(sourceId)));
89  }
90 
91  wsm->setDestinyNode(prepareICRNode(oIcarQoc->oIcarContext->agentGroup->getRemoteAgent(destinationId)));
92 
93  wsm->setNextNode(prepareICRNode(oIcarQoc->oIcarContext->agentGroup->getRemoteAgent(nextId)));
94 
95  //sendown message
96  sendICRMessage(wsm, "transmitting", "msgIcar");
97  return(wsm);
