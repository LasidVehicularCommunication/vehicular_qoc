74 void AgentPairMobility::calcRelativeSpeed(){
75  double angleS = c->getAgentS()->getMobilityInfo()->getAngle();
76  double angleD = c->getAgentD()->getMobilityInfo()->getAngle();
77 
78  double vxS=c->getAgentS()->getMobilityInfo()->getActualMove()->getSpeed()* std::cos(angleS);
79  double vyS=c->getAgentS()->getMobilityInfo()->getActualMove()->getSpeed()* std::sin(angleS);
80  double vxD=c->getAgentD()->getMobilityInfo()->getActualMove()->getSpeed()* std::cos(angleD);
81  double vyD=c->getAgentD()->getMobilityInfo()->getActualMove()->getSpeed()* std::sin(angleD);
82  relativeSpeedX =vxS-vxD;
83  relativeSpeedY =vyS-vyD;
84  relativeSpeed = std::sqrt((relativeSpeedX*relativeSpeedX) + (relativeSpeedY*relativeSpeedY));
85  relativeSpeedAngleRad = std::atan2(relativeSpeedY, relativeSpeedX); //Relative Velocity angle
86 
87  //Max Relative Speed
88  this->maxRelativeSpeed = c->getAgentS()->getMobilityInfo()->getMaxSpeed()
89  + c->getAgentD()->getMobilityInfo()->getMaxSpeed();
90 
91  double vxSMax=c->getAgentS()->getMobilityInfo()->getMaxSpeed() * std::cos(angleS);
92  double vySMax=c->getAgentS()->getMobilityInfo()->getMaxSpeed() * std::sin(angleS);
93  double vxDMax=c->getAgentD()->getMobilityInfo()->getMaxSpeed() * std::cos(angleD);
94  double vyDMax=c->getAgentD()->getMobilityInfo()->getMaxSpeed() * std::sin(angleD);
95  double relativeSpeedXMax =vxSMax-vxDMax;
96  double relativeSpeedYMax =vxSMax- vyDMax;
97  this->maxAngleRelativeSpeed = std::sqrt((relativeSpeedXMax*relativeSpeedXMax) + (relativeSpeedYMax*relativeSpeedYMax));
98 }

100 void AgentPairMobility::calcDistance(){
101  //Calc vehicles distance
102  Coord s = c->getAgentS()->getMobilityInfo()->getActualMove()->getStartPos();
103  Coord d = c->getAgentD()->getMobilityInfo()->getActualMove()->getStartPos();
104 
105  this->distancelrX = (s.x - d.x);
106  this->distancelrY = (s.y - d.y);
107  this->distancelrZ = (s.z - d.z);
108  //this->distancelrZ = 0;
109  this->distancelr = std::sqrt((this->distancelrX*this->distancelrX) + (this->distancelrY*this->distancelrY));
110  this->distancelrAngleRad = std::atan2(this->distancelrY,this->distancelrX); //rl Distance angle
111 }
