169 void Icarqoc::handleSelfMsg(cMessage* msg) {
170  //BaseWaveApplLayer::handleSelfMsg(msg);
171  //this method is for self messages (mostly timers)
172  //it is important to call the BaseWaveApplLayer function for BSM and WSM transmission
173  //SEND_ICM_EVT, // general event of IC Message
174  //SEND/-*--_ICAR_PROTOCOL, //Data dissemination
175  //SEND_FORMATION_VEHIC, // vehicular formation
176  //SEND_IMAGES_STREAMS // images streams
177  //MEASURE_COMM_PERFORMANCE // event to measure the environment communication performance
178  //AGENTPAIR_MONITOR_MESSAGE = For each knowed CHANNEL when need to update communication state (communicatin, non-communicating, fault etc.)
179 
180  // verify the simulation time.
181  if (simTime().dbl() <= this->oKnownGlobal->finishSimulation)
182  {
183  switch (msg->getKind())
184  {
185  case MEASURE_COMM_PERFORMANCE:
186  {
187  // calc performance measurement
188  //if (msg->getName()=="firstLoad") this->oVision->calcCommPerformVision(true);
189  //else this->oVision->calcCommPerformVision(false);
190 
191  //schedule new performance measurement
192  //this->scheduleLoad(this->perforMesurementPeriod, MEASURE_COMM_PERFORMANCE, "commperform", msg);
193  }break;
194 
195  case AGENTPAIR_MONITOR_MESSAGE:
196  {
197  AgentPair * c = (AgentPair *)(msg->getContextPointer());
198  int64_t s = c->getAgentS()->getId();
199  int64_t d = c->getAgentD()->getId();
200  //oVision->updateCommunicationState(s,d);
201  }break;
202 
203  case COLLECT_MOBILITY:
204  {
205  //std::cout << endl << "begin schedule new mobility collect "<< endl;
206 
207  //schedule new mobility collect
208  if (oKnownGlobal->mobilityPeriod == -1){
209  double tempx = (double(rand()%100+1)/10)/2.5;
210  this->scheduleLoad(tempx, COLLECT_MOBILITY, "collectMobility", msg);
211  //std::cout << endl << "range Mobility= "<< tempx << endl;
212  }
213 
214  if (oKnownGlobal->mobilityPeriod >0)
215  this->scheduleLoad(this->oKnownGlobal->mobilityPeriod, COLLECT_MOBILITY, "collectMobility", msg);
216 
217  this->changeLocalMobility();
218 
219  //std::cout << endl << "end schedule new mobility collect "<< endl;
220 
221  }break;
222 
223  case SEND_MONITOR_EVT:
224  {
225  //std::cout << endl << "begin send monitor message "<< endl;
226 
227  //send monitor message
228  loadMonitor(msg);
229 
230  //std::cout << endl << "end send monitor message "<< endl;
231 
232  }break;
233 
234  case SEND_ICM_EVT:
235  {
236  //std::cout << endl << "begin send load Application message "<< endl;
237  loadApp(msg);
238  this->scheduleLoad(this->loadPeriodApp, SEND_ICM_EVT, "appMessage", msg);
239  //std::cout << endl << "end send load Application message "<< endl;
240 
241  }break;
242 
243 
244  default:
245  {
246  if (msg)
247  EV << "APP: Error: Got Self Message of unknown kind! Name: " << msg->getName() << endl;
248  }break;
249 
250  }//end switchcase
251  }//endif
252 }
